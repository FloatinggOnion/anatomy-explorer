       (using .current to avoid recreating on each render — same trick as SceneController's ref guards)

    useFrame(({ camera }) => {
      const ndc = pointingNDCRef.current
      const group = modelGroupRef.current
      if (!group) return

      if (!ndc) {
        // No pointing gesture — reset dwell but do NOT clear selectedMeshName
        // (label stays until user actively points elsewhere or at empty space)
        dwellStartRef.current = null
        dwellMeshRef.current = null
        return
      }

      raycaster.setFromCamera(ndc, camera)
      const hits = raycaster.intersectObject(group, true) // recursive=true

      // Find the first named ancestor group within modelGroupRef
      const hitMesh = hits.length > 0 ? hits[0].object : null
      const meshName = hitMesh ? findNamedAncestor(hitMesh, group) : null

      // Dwell logic (Pattern 5 from RESEARCH.md):
      if (meshName === dwellMeshRef.current) {
        if (dwellStartRef.current !== null) {
          if (performance.now() - dwellStartRef.current >= DWELL_MS) {
            // Dwell complete — update Zustand (null = dismiss, string = select)
            setSelectedMeshName(meshName)
            dwellStartRef.current = null // reset to prevent re-firing
          }
          // else: still waiting
        } else {
          dwellStartRef.current = performance.now() // start timer
        }
      } else {
        // Target changed — reset timer
        dwellMeshRef.current = meshName
        dwellStartRef.current = performance.now()
      }
    })

    findNamedAncestor helper function (module-level):
    Walk object.parent chain up toward root (modelGroupRef.current).
    Return the first ancestor whose name is non-empty string.
    Stop when current.parent === root OR current === root.
    This handles the two-level depth of SkeletonPreview (root → unnamed ProceduralSkeleton → named groups):
      function findNamedAncestor(obj: THREE.Object3D, root: THREE.Object3D): string | null
        let current: THREE.Object3D | null = obj
        while (current && current !== root) {
          if (current.name && current.name.length > 0) return current.name
          current = current.parent
        }
        return null

    Return null from the component (invisible controller).
