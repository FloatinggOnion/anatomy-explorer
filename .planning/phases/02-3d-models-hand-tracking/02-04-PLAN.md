    interpret = useCallback((landmarks: NormalizedLandmark[][], videoWidth: number, videoHeight: number): GestureCommand => {
      setHandDetected(landmarks.length > 0);
      if (landmarks.length === 0) {
        // No hands: clear gesture state, debounce gesture off (D-25)
        gestureStateRef.current = { mode: 'idle', pinchOrigin: null };
        isPinchingRef.current = false;
        if (gestureOffTimerRef.current) clearTimeout(gestureOffTimerRef.current);
        gestureOffTimerRef.current = setTimeout(() => setGestureActive(false), 500);
        return { type: 'idle' };
      }

      // Single-hand pinch detection (landmarks[0]):
      const hand0 = landmarks[0];
      const thumbTip = hand0[4]; const indexTip = hand0[8];
      const pinchDist0 = Math.hypot(thumbTip.x - indexTip.x, thumbTip.y - indexTip.y);
      const wasPinching = gestureStateRef.current.mode === 'pinching' || gestureStateRef.current.mode === 'two-hand-pinch';
      const threshold = wasPinching ? PINCH_EXIT : PINCH_ENTER;
      const isPinching0 = pinchDist0 < threshold;

      // Two-hand detection:
      if (landmarks.length >= 2 && isPinching0) {
        const hand1 = landmarks[1];
        const thumbTip1 = hand1[4]; const indexTip1 = hand1[8];
        const pinchDist1 = Math.hypot(thumbTip1.x - indexTip1.x, thumbTip1.y - indexTip1.y);
        const isPinching1 = pinchDist1 < threshold;

        if (isPinching1) {
          // Both hands pinching — compute pinch center for each hand
          const pinchCenter0 = { x: (thumbTip.x + indexTip.x) / 2 * videoWidth, y: (thumbTip.y + indexTip.y) / 2 * videoHeight };
          const pinchCenter1 = { x: (thumbTip1.x + indexTip1.x) / 2 * videoWidth, y: (thumbTip1.y + indexTip1.y) / 2 * videoHeight };
          const currentDist = Math.hypot(pinchCenter0.x - pinchCenter1.x, pinchCenter0.y - pinchCenter1.y);

          if (gestureStateRef.current.mode !== 'two-hand-pinch') {
            // Just entered two-hand mode — store initial distance; reset midpoint tracker
            gestureStateRef.current = { mode: 'two-hand-pinch', pinchOrigin: { x: currentDist, y: 0 } };
            prevMidpointRef.current = null; // reset so first frame doesn't emit a spurious pan
            setGestureActive(true); if (gestureOffTimerRef.current) clearTimeout(gestureOffTimerRef.current);
            return { type: 'idle' };
          }

          const prevDist = gestureStateRef.current.pinchOrigin!.x;
          const midpoint0 = { x: (pinchCenter0.x + pinchCenter1.x) / 2, y: (pinchCenter0.y + pinchCenter1.y) / 2 };

          // Disambiguate pan vs scale (D-20): same-direction movement = pan; apart/together = scale.
          // Use a second ref prevMidpointRef to track the previous midpoint for pan delta.
          // If prevMidpointRef.current is null (just entered two-hand mode), it was set to null above — skip.
          const scaleFactor = currentDist / Math.max(prevDist, 1);
          gestureStateRef.current.pinchOrigin = { x: currentDist, y: 0 };
          isPinchingRef.current = true;

          // Pan detection: compare current midpoint to previous midpoint stored in prevMidpointRef.
          // prevMidpointRef is a separate useRef<{ x: number; y: number } | null>(null) added to the hook.
          if (prevMidpointRef.current !== null) {
            const panDelta = {
              x: midpoint0.x - prevMidpointRef.current.x,
              y: midpoint0.y - prevMidpointRef.current.y,
            };
            const panMagnitude = Math.hypot(panDelta.x, panDelta.y);
            // Scale magnitude is how much the distance changed (ratio deviation from 1.0)
            const scaleMagnitude = Math.abs(scaleFactor - 1.0);
            // If pan movement exceeds scale change and dead zone, emit pan instead of scale.
            // Threshold: pan dominates when panMagnitude > DEAD_ZONE_PX and panMagnitude > scaleMagnitude * 200.
            if (panMagnitude > DEAD_ZONE_PX && panMagnitude > scaleMagnitude * 200) {
              prevMidpointRef.current = midpoint0;
              return { type: 'pan', delta: panDelta };
            }
          }
          prevMidpointRef.current = midpoint0;
          return { type: 'scale', factor: scaleFactor };
        }
      }

      // Single-hand pinch:
      if (isPinching0) {
        const handCenter = { x: hand0[9].x * videoWidth, y: hand0[9].y * videoHeight };
        if (gestureStateRef.current.mode !== 'pinching') {
          // Pinch just started
          gestureStateRef.current = { mode: 'pinching', pinchOrigin: handCenter };
          isPinchingRef.current = true;
          setGestureActive(true); if (gestureOffTimerRef.current) clearTimeout(gestureOffTimerRef.current);
          return { type: 'idle' };
        }
        const origin = gestureStateRef.current.pinchOrigin!;
        const delta = { x: handCenter.x - origin.x, y: handCenter.y - origin.y };
        const magnitude = Math.hypot(delta.x, delta.y);
        if (magnitude < DEAD_ZONE_PX) return { type: 'idle' }; // D-19
        // Update origin for incremental delta (trackball accumulates)
        gestureStateRef.current.pinchOrigin = handCenter;
        isPinchingRef.current = true;
        return { type: 'rotate', delta: { x: delta.x * rotationSensitivity, y: delta.y * rotationSensitivity } };
      }

      // Was pinching, now not:
      gestureStateRef.current = { mode: 'idle', pinchOrigin: null };
      isPinchingRef.current = false;
      if (gestureOffTimerRef.current) clearTimeout(gestureOffTimerRef.current);
      gestureOffTimerRef.current = setTimeout(() => setGestureActive(false), 500); // D-25
      return { type: 'idle' };
    }, [setGestureActive, setHandDetected, PINCH_ENTER, PINCH_EXIT, DEAD_ZONE_PX, rotationSensitivity]);

    Return { interpret, isPinching: isPinchingRef.current }. Note: isPinchingRef.current is not reactive — callers in App.tsx should use a useState boolean updated in the onResults callback for reactive LandmarkCanvas highlighting.

    useFrame(() => {
      const group = modelGroupRef.current;
      if (!group) return; // Model not yet loaded — skip frame
      const cmd = gestureCommandRef.current;

      // Apply momentum decay first (300ms deceleration, Claude's discretion per UI-SPEC)
      if (isDeceleratingRef.current) {
        velocityRef.current.x *= 0.92; velocityRef.current.y *= 0.92;
        const speed = Math.hypot(velocityRef.current.x, velocityRef.current.y);
        if (speed < 0.0001) { isDeceleratingRef.current = false; velocityRef.current = { x: 0, y: 0 }; }
